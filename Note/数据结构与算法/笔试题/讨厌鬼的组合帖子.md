import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        
        Scanner scanner = new Scanner(System.in);
        int num = scanner.nextInt();
        int[] a = new int[num];
        for(int i = 0; i < num; i++){
            a[i] = scanner.nextInt();
        }
        int[] b = new int[num];
        for(int i = 0; i < num; i++){
            b[i] = scanner.nextInt();
        }
        
        long maxPos = 0;
        long maxNeg = 0;
        for(int i = 0; i < num; i++){
            if(a[i] >= b[i]){
                maxPos += a[i] - b[i];
            }
            else{
                maxNeg += b[i] - a[i];
            }
        }
        System.out.println(Math.max(maxPos, maxNeg));

        scanner.close();
    }
}
